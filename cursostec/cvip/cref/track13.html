13.01 asctime
        Esta função converte data e tempo, estocados numa struct tm, para string.
        
        char *asctime(const struct tm *tblock);
        sMsg = asctime(&xGmt);

        sMsg......A função retorna um ponteiro para a string obtida

        Veja este exemplo de uso:
        #include "stdio.h"
        #include "conio.h"
        #include "time.h"

        int main(void) {
        struct tm xGmt;
        char str[80];
        char *sMsg;

        clrscr();

        xGmt.tm_sec    = 1;   /* Segundos                                      */
        xGmt.tm_min    = 30;  /* Minutos                                       */
        xGmt.tm_hour   = 9;   /* Horas                                         */
        xGmt.tm_mday   = 1;   /* Dia do mês                                    */
        xGmt.tm_mon    = 2;   /* Mês                                           */
        xGmt.tm_year   = 105; /* Ano (Este valor será somado com 1900)         */
        xGmt.tm_wday   = 5;   /* Dia da semana                                 */
        xGmt.tm_yday   = 0;   /* Não utilizado por asctime()                   */
        xGmt.tm_isdst  = 0;   /* Horário de verão, não utilizado por asctime() */

        sMsg = asctime(&xGmt);
        printf("%s\n", sMsg);

        puts("End of Program.");
        getch();
        return 0;}

        Saída:
        Fri Mar 01 09:30:01 2005
        End of Program.

13.02 clock
        Esta função retorna o número de clock ticks decorridos desde o início
        do programa. Para converter o valor de retorno para segundos basta
        dividir este valor pela constante CLK_TCK.
        
        clock_t clock(void);
        nTicks = clock();

        Ex.:    clock_t nTicks;
                nTics = clock();

        Veja este exemplo:
        #include "time.h"
        #include "stdio.h"
        #include "dos.h"
        #include "conio.h"

        int main(void) {
        clock_t start, end;

        clrscr();

        start = clock();

        delay(2000);

        end = clock();
        printf("O tempo decorrido foi %f segundos.\n", (end - start) / CLK_TCK);
        getch();
        return 0; }

        O tempo decorrido foi 0.109890 segundos.

13.03 ctime
        Esta função converte data e tempo para string a partir de nSeconds
        que é o retorno da funçao time().
        
        char *ctime(const time_t *time);
        sHoras = ctime(&nSeconds);

        Ex.:    char *sHoras;
                time_t nSeconds;
                nSeconds = time();
                sHoras = ctime(&nSeconds);
                printf("%s\n",sHoras);
                /* Resulta em   Sat Jan 22 10:54:17 2005 */

13.04 difftime
        Esta função calcula a diferença em segundos entre dois pontos.
        
        double difftime(time_t time2, time_t time1);
        nDeltaTempo = difftime(nFinal,nInicial);

        Ex.:    double nDeltaTempo;
                time_t nFinal;
                time_t nInicial;
                nInicial = time(NULL);
                sleep(2);
                nFinal = time(NULL);
                nDeltaTempo = difftime(nFinal, nInicial);
                printf("A diferença é %f segundos.\n",nDeltaTempo);
                /* Resulta em A diferença é 2.000000 segundos. */

13.05 gmtime
        Esta função converte data e tempo para o padrão GMT (GreenWich Mean Time).
        A entrada para esta conversão é nSeconds retornado por time(), a saída
        desta função é um ponteiro para uma estrutura tm.
        
        struct tm *gmtime(const time_t *timer);
        xGmt = gmtime(&nSeconds);

        Elementos da struct tm:
        --------------------------------------------------------------
        int xGmt.tm_hour.....horas
        int xGmt.tm_min......minutos
        int xGmt.tm_sec......segundos
        int xGmt.tm_mday.....dia do mês
        int xGmt.tm_mon......mês (-1)
        int xGmt.tm_year.....ano (-1900)
        int xGmt.tm_wday.....dia da semana (-1)
        int xGmt.tm_isdst....Indica horário de verão

        Ex.:    struct tm *xGmt;
                time_t nSeconds;
                time(&nSeconds);
                xGmt = gmtime(&nSeconds);

        Veja programa exemplo:
        #include "time.h"
        #include "stdio.h"
        #include "conio.h"

        int main(void) {

        struct tm *xGmt;
        time_t nSeconds;

        clrscr();

        nSeconds = time(&nSeconds);

        printf ("%li\n",nSeconds);
        xGmt = gmtime(&nSeconds);

        printf ("xGmt.tm_hour: %d\n",xGmt->tm_hour);
        printf ("xGmt.tm_min: %d\n",xGmt->tm_min);
        printf ("xGmt.tm_sec: %d\n",xGmt->tm_sec);
        printf ("xGmt.tm_mday: %d\n",xGmt->tm_mday);
        printf ("xGmt.tm_mon: %d\n",xGmt->tm_mon+1);
        printf ("xGmt.tm_year: %d\n",xGmt->tm_year+1900);
        printf ("xGmt.tm_wday: %d\n",xGmt->tm_wday+1);
        printf ("xGmt.tm_isdst: %d\n",xGmt->tm_isdst);

        getch();

        puts("End of Program");
        getch(); }

        Saída:
        1106413483
        xGmt.tm_hour: 17
        xGmt.tm_min: 4
        xGmt.tm_sec: 43
        xGmt.tm_mday: 22
        xGmt.tm_mon: 1
        xGmt.tm_year: 2005
        xGmt.tm_wday: 7
        xGmt.tm_isdst: 0

13.07 stime
        Esta função configura data e hora do sistema em termos de nSegundos
        depois do dia 01/01/1970 0:00hs.
        
        int stime(time_t *p);
        stime(&nSeconds);

        Ex.:    time_t nSeconds;
                nSeconds = time(NULL):
                nSeconds += 24L*60L*60L
                stime(&nSeconds);
                /* Avançamos aqui a data em um dia */

        Veja este programa:
        #include "time.h"
        #include "stdio.h"
        #include "conio.h"

        int main(void) {
        time_t nSeconds,nMem;

        clrscr();

        nSeconds = time(NULL);
        nMem = nSeconds;
        printf("Hoje é %s", ctime(&nSeconds));

        nSeconds -= 24L*60L*60L;  /* Vamos retroceder a data em um dia */
        stime(&nSeconds);
        printf("\nA nova data é %s", ctime(&nSeconds));

        puts("End of Program");
        stime(&nMem);
        getch();}

        Saída:
        Hoje é Sat Jan 22 15:05:09 2005

        A nova data é Fri Jan 21 15:05:09 2005
        End of Program

13.08 time
        Esta função obtém o número de segundos que transcorreu desde
        01/01/1970.
        
        time_t time(time_t *timer);
        nSeconds = time(&nSeconds);

        nSeconds.....A função retorna um ponteiro para nSeconds

        Ex.:    time_t nSeconds;
                time(&nSeconds);

        Veja este programa exemplo:
        #include "stdlib.h"
        #include "stdio.h"
        #include "conio.h"
        #include "time.h"
        #include "stdio.h"
        #include "dos.h"

        int main(void){
        time_t nSeconds;

        time(&nSeconds);

        clrscr();
        printf("Número de segundos desde 01/01/1970 é %ld\n",nSeconds);

        puts("End of Program.");
        getch();
        return 0;}

        Saída:
        Número de segundos desde 01/01/1970 é 1106274725
        End of Program.

13.09 tzset
        Esta função cria um ambiente para compatibilidade com UNIX
        na formatação de tempo.
        
        void tzset(void);
        tzset();

        Ex.:    tzset();

        *** Para uso desta função você deve criar uma variável de ambiente
        nomeada TZ que tzset() utilizará para seus processos de compatibilização
        com o sistema UNIX. Consulte documentação do Turbo C++ da Borland para
        uma descrição detalhada de como configurar TZ.

        Veja este programa exemplo:
        #include "time.h"
        #include "stdio.h"
        #include "conio.h"
        #include "stdlib.h"

        int main(void){
        time_t td;
        clrscr();

        putenv("TZ=PST8PDT");
        tzset();
        time(&td);
        printf("Data corrente = %s\n", asctime(localtime(&td)));
        getch();
        return 0;}
