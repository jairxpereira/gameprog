14.09 cos

14.15 frexp
        Esta função decompõe o valor fornecido em sua mantissa e expoente
        que lhe dão origem, de forma que este valor seja reobtido através da
        seguinte fórmula: nNumber = nResult * 2N
        
        double frexp(double value, int *exponent);
        nResult = frexp(nNumber, &nExpoente);

        Ex.:    double nResult;
                int nExpoente;
                double nNumber = 8.0;
                nResult = frexp(nNumber, &nExpoente);
                printf("nResult: %lf \t nExpoente: %d\n");
                /* Resulta em nResult: 0.50000  nExpoente: 4 */

                Perceba então que:
                nNumber = 0.5 * 24
                nNumber = 0.5 * 16
                nNumber = 8

        Veja mais este programa exemplo:
        #include "stdio.h"
        #include "conio.h"
        #include "math.h"

        int main(void) {
        double nMantissa, nNumber;
        int nExpoente;

        nNumber = 8.0;
        nMantissa = frexp(nNumber, &nExpoente);

        clrscr();
        printf("O nro %lf ‚ ", nNumber);
        printf("%lf * 2 a ", nMantissa);
        printf("elevado a %d\n", nExpoente);

        getch();
        return 0; }

        Saída:
        O nro 8.000000 é 0.500000 * 2 elevado a 4.

14.16 hypot
        Esta função calcula a hipotenusa de um triângulo retângulo a partir da
        medida dos catetos.

        nResult = hypot(xCat, yCat);

        Ex.:    double nResult, xCat=4, yCat=3;
                nResult = hypot(xCat, yCat);
                printf ("%lf\n",nResult);
                /* Resulta em 5.000000 */

        A fórmula do triângulo retângulo é esta: hypot2 = xCat2 + yCat2
        x2 = 42 * 32
        x2 = 16 + 9
        x2 = 25
        sqrt(25) = 5
        x = 5

14.21 matherr
        Esta função é uma interface para o usuário definir sua própria rotina
        de manipulação de erros das funções matemáticas. Esta função é chamada
        naturalmente sempre que ocorre um erro nesta funções.
        
        int matherr (struct exception *e);
        nSts = matherr(xMathTrap);

        Elementos da struct exception:
        ------------------------------------------------------------
        int xMathTrap.type........tipo do erro ocorrido
        char xMathTrap.name.......nome da função de origem do erro
        double xMathTrap.arg1.....primeiro argumento da função
        double xMathTrap.arg2.....segundo argumento da função
        double retval.............valor de retorno da função

        Veja um exemplo de redefinição de matherr(): 
        #include "stdio.h"
        #include "conio.h"
        #include "math.h"
        #include "string.h"

        int matherr (struct exception *xMathTrap) {
        if (xMathTrap->type == DOMAIN)
        if (!strcmp(xMathTrap->name,"sqrt")) {
        xMathTrap->retval = sqrt (-(xMathTrap->arg1));
        return 1; }
        return 0; }

        int main(void) {
        double x = -2.0, y;

        clrscr();
        y = sqrt(x);
        printf("Matherr corrigiu o valor: %lf\n",y);
        getch();
        return 0;}

14.23 poly
        Esta função gera um polinômio a partir dos argumentos, e retorna o
        valor desse cálculo baseado no valor dado de x.
        
        double poly (double x, int degree, double coeffs[]);
        nResult = poly (x, nGrau, nCoeficientes);

        Ex.: /* x3 - 2x2 + 5x - 1 com x=2*/
        double nResult;
        double x = 2.0;
        double nGrau = 3;
        double nCoeficientes[] = { -1.0, 5.0, -2.0, 1.0};
        nResult = poly(x, nGrau, nCoeficientes);
        printf ("%lf\n",nResult);
        /* Resulta em 9.000000 */

        Perceba que:
        x3 - 2x2 + 5x - 1 com x=2
        8 - 8 + 10 - 1 = 9

        Outro detalhe, perceba que a array de coeficientes os recebem em
        ordem inversa de chegada na equação:
        x3 - 2x2 + 5x - 1 com x=2
        double nCoeficientes[] = { -1.0, 5.0, -2.0, 1.0};

        Veja este programa exemplo: 
        #include "stdio.h"
        #include "conio.h"
        #include "math.h"

        int main(void) {

        char sPoly[]="x**3 - 2x**2 + 5x - 1";

        double nCoeficientes[] = { -1.0, 5.0, -2.0, 1.0};
        double nResult;
        double x = 2.0;
        double nGrau = 3;

        clrscr();
        nResult = poly(x, nGrau, nCoeficientes);
        printf("%s com x=2 ‚ %lf\n",sPoly,nResult);

        getch();
        return 0; }
