05-2. Modificadores de acesso
1. Acesso público: public     

// Definição do atributo nvezes na classe Ferramentas
public int nvezes = 0;

O membro definido como public pode ser visto, acessado e modificado
por outras classes. Veja no código abaixo que acessamos o atributo
nvezes sem nenhuma restrição na classe Program:

 // Chamada ocorrida na classe Program
 Console.WriteLine(" Contagem de uso das ferramentas: " +
        Ferramentas.nvezes);   
		
No contexto de herança de classes, os membros públicos são herdados
pelas classes filhas.

2. Acesso particular: private

// Definição do atributo nvezes na classe Ferramentas
private int nvezes = 0;
O membro definido como private somente pode ser acessado
e modificado pelos métodos internos da própria classe. 

Altere na classe Ferramentas, do projeto tstStatic, o modificador de nvezes
para private e veja que o programa deixa de funcionar devido ao conflito de
nível de acesso.



Numa solução mais elegante, o ideal é configurar o atributo como private
e prover métodos públicos para recuperá-lo (get) ou configurá-lo (set).
Dessa forma, o programa fica  mais  seguro  porque  os  valores  sendo
configurados no  atributo podem ser verificados pelo código interno do
método.   É   essa   idéia  de elegância e segurança que deu origem às 
propriedades que vamos ver mais adiante no curso.


No contexto de herança, os membros particulares (private) não são
herdados pelas classes filhas. 

3. Acesso protegido: protected

// Definição do atributo nvezes na classe Ferramentas
protected int nvezes = 0;
O membro definido como protected somente pode ser acessado
e modificado pelos métodos internos da própria classe e também 
pelas classes filhas no contexto de herança. 

Altere na classe Ferramentas, do projeto tstStatic, o modificador de nvezes
para protected e veja que o programa deixa de funcionar devido ao conflito de
nível de acesso. 



O programa deixa de funcionar também porque classes estáticas não podem
ter membros protegidos visto que classes estáticas não podem participar
do processo de herança.



Veja no tópico abaixo a listagem do código fonte do projeto
prj_modificadores  para  fixar  o  uso   dos  modificadores
private e public.

3. Listagem do código do projeto prj_modificadores




// projeto prj_modificadores - Arquivo: Program.cs
// Esse projeto ilustra o uso de modificadores de acesso
using System;

namespace prj_modificadores
{
  class Program
  {

    static void Main(string[] args)
    {
      // tools é um objeto da classe Ferramentas
      Ferramentas tools = new Ferramentas();

      tools.config_janela("projeto prj_modificadores");

      Console.WriteLine("\n Nome do usuário: " +
        tools.nome_usuario());

      Console.WriteLine(" Nome do computador: " +
        tools.nome_computador());

      Console.WriteLine(" Contagem de uso das ferramentas: " +
      tools.get_contador());

      tools.set_contador(5);

      tools.mostrar_contador();  

      Console.Read();

    } // Main().fim
  } // class.fim
} // namespace.fim




// projeto prj_modificadores - Arquivo: Ferramentas.cs
using System;

namespace prj_modificadores
{
  class Ferramentas
  {

    static private int nvezes = 0;

    // Retorna o nome de rede do computador    
    public String nome_computador()
    {
      String nome_pc = "";
      nome_pc = Environment.MachineName;
      nvezes++;
      return nome_pc;
    } // nome_computador().fim

    // Retorna o nome do usuário local
    public String nome_usuario()
    {
      String usuario = "";
      usuario = Environment.UserName;
      nvezes++;
      return usuario;
    } // nome_usuario().fim

    // Método para configurar a janela
    public void config_janela(String titulo)
    {
      // Vamos configurar a janela
      Console.ForegroundColor = ConsoleColor.Blue;
      Console.BackgroundColor = ConsoleColor.White;
      Console.Title = titulo;
      Console.SetBufferSize(80, 30);
      Console.SetWindowSize(80, 30);
      Console.Clear();
      nvezes++;
    } // config_janela() fim    

    
    // Retorna o contador
    public int get_contador()
    {
      return nvezes;
    }

    // Resseta o contador
    public void set_contador( int nval)
    {
    nvezes = nval;
    Console.WriteLine("\n\n O contador de uso foi modificado");
    }

    // Mostra o contador
    public void mostrar_contador()
    {
      Console.WriteLine ( " Contador: {0}", nvezes);
    }

  } // fim da classe
} // fim do namespace
