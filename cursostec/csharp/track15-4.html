1.1 A aplicação cliente


// Recebe e envia dados via tcp\ip
private static void iniciar_cliente()
{
  // Memória para receber os dados
  string dados_chegando = "";
  string sComando = "";

  char[] memoria = new char[256];
  int ntam = memoria.Length;

  // Cria uma tomada TcpCliente para conversar com o servidor
  TcpClient tomada_servidor;
  
  try
  {
    tomada_servidor = new TcpClient("localHost", 65000);
  }
  catch
  {
    Console.WriteLine(
    " Falha de conexão com o servidor em {0}:65000",
    "localhost");
    return;
  }

  // Cria o stream de network, fluxo_rede, e o stream de leitura
  // e o stream escritor.
  NetworkStream fluxo_rede = tomada_servidor.GetStream();
  StreamReader leitor = new StreamReader(fluxo_rede);
  StreamWriter escritor = new StreamWriter(fluxo_rede);

  // Quando falso encerra o processo de leitura e gravação
  bool continuar = true;

  Console.WriteLine(" Digite 'parar' para finalizar. ");
  
  while (continuar == true)
  {
    // Recebe um comando do usuário
    Console.Write(" >");
    sComando = Console.ReadLine();
    if (sComando.Equals("parar")) continuar = false;

    try
    {                   
      // Transmite para o Servidor
      escritor.WriteLine(sComando);
      escritor.Flush();

    } // endtry

    // Verifica erros de gracação
    catch (Exception erro)
    {
      Console.WriteLine(
      " Erro de gravação ao servidor:\n {0}", erro.Message);
    } // endcatch

    
    // Recepção de dados do servidor
    try
    {
      int nlidos = leitor.Read(memoria, 0, ntam);
      if (nlidos > 0)
      {
        dados_chegando = montarTexto(memoria);
        Console.WriteLine(" {0}", dados_chegando);
      } // endif
    } // endtry

    // Verifica erros de leitura
    catch (Exception erro)
    {
      Console.WriteLine("Erro de leitura do servidor:\n {0}", erro.Message);
     } // endcatch

  }// endwhile

  // Fechando a conexão
  fluxo_rede.Close();
  leitor.Close();
  escritor.Close();
  } // iniciar_cliente().fim 



// Memória para receber os dados
string dados_chegando = "";
string sComando = "";
As strings são para tornar confortável a vida do usuário  do  programa  que
vai  trabalhar recebendo e enviando textos legíveis. sComando vai receber o
comando digitado  pelo usuário e dados_chegando vai receber a resposta dada
pelo servidor ao comando enviado a ele.

char[] memoria = new char[256];
int ntam = memoria.Length;
// (...) int nlidos = leitor.Read(memoria, 0, ntam);
A variável memoria é para suportar os dados lidos  do  fluxo  de  rede  que
estarão formatados no tipo char. 

// Cria uma tomada TcpCliente para conversar com o servidor
TcpClient tomada_servidor;
tomada_servidor = new TcpClient("localHost", 65000);
Aqui é o código que estabelece a conexão com a aplicação servidora.
É importante cercar esse código com try\catch para pegar erros que
ocorram na conexão.

// Cria o stream de network, fluxo_rede, e o stream de leitura
// e o stream escritor.
NetworkStream fluxo_rede = tomada_servidor.GetStream();
StreamReader leitor = new StreamReader(fluxo_rede);

Após o estabelecimento da conexão são criados os objetos para  manipular  o 
fluxo de dados.

// Recebe um comando do usuário
Console.Write(" >");
sComando = Console.ReadLine();
if (sComando.Equals("parar")) continuar = false;
Esse código recebe o comando do usuário; se o usuário digitar 'parar', 
isso quebra o laço do bloco while e fecha os streams abertos.

// Transmite para o Servidor
escritor.WriteLine(sComando);
escritor.Flush();
Aqui enviamos o comando para o servidor.

// Recepção de dados do servidor
int nlidos = leitor.Read(memoria, 0, ntam);
dados_chegando = montarTexto(memoria);
Console.WriteLine(" {0}", dados_chegando);
Esse código recebe a resposta do servidor que vem na forma de uma array do 
tipo  char  que  a  função  montarTexto transforma numa string legível aos
olhos do usuário.

// Fechando a conexão
fluxo_rede.Close(); leitor.Close(); escritor.close();
Aqui fazemos o saneamento do ambiente fechando os streams abertos.


1.2 Transformando bytes em texto


// Converte uma char[] em string
private static string montarTexto ( char[] dados)
{
  string txt = "";
  byte[] txt_bytes;
  txt_bytes = Encoding.UTF8.GetBytes(dados);
  txt = Encoding.UTF8.GetString(txt_bytes);
  txt = txt.Trim();
  return txt ;
} // montarTexto().fim



Aqui transformamos uma array de char em bytes para depois transformá-los
numa string.
