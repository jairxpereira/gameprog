07-1 Criando interfaces
1. Visão geral
A interface define propriedades e métodos que devem ser obrigatoriamente
implementados pela classe que a herda. 

2. Construindo uma interface


interface IMonstro
{

    // Interfaces não podem ter campos (atributos), apenas 
    // propriedades e métodos.

    // Todo monstro IMonstro tem que ter nome e fazer estrago!
    int Estrago   { get; set; } // capacidade de fazer estrago
    string Nome  { get; set; }  // nome do objeto   
    
    // atacar()
    void atacar();

} // fim da classe


Neste pequeno exemplo mostramos como construir uma interface. Estabelecemos
que  todo  objeto  cuja  classe  herdou  nossa  interface deve ter o método
atacar()  e  a  propriedade  Estrago.  Veja  no exemplo abaixo a sintaxe de 
herança de uma interface por uma classe:



 class Ogro : IMonstro
  {
    // campos
    private int m_estrago;   // capacidade de fazer estrago
    private string m_nome;   // nome do objeto   
   
    // Construtor 
    public Ogro() 
    {
      m_estrago = 120; 
      m_nome = "ogro";
    }

    // Vamos implementar os membros conforme contrato com IMonstro
    public void atacar()
    {
      string msg = "";
      Console.Write("\n");

      msg = " (Classe Ogro) {0} ataca.";
      Console.Write(msg, m_nome);
      msg = " Vc perde {0} pontos de energia.";
      Console.WriteLine(msg, m_estrago);
    }// atacar().fim

    // get/set sobre m_estrago
    public int Estrago
    {    
     set { m_estrago = value; }
     get { return m_estrago; }    
    } // fim da propriedade;

    // get/set sobre m_nome
    public string Nome
    {
      set { m_nome = value; }
      get { return m_nome; }
    } // fim da propriedade;

 } // fim da classe
 

 
 3. Herança múltipla de Interface
 Ao contrário do que ocorre entre as classes, no uso de interfaces pode
 ocorrer herança múltipla. Uma classe pode herdar definições de duas ou
 mais interfaces. 
 
 


interface IMonstro
{
    // Todo monstro IMonstro tem que ter nome e fazer estrago!
    int Estrago   { get; set; } // capacidade de fazer estrago
    string Nome  { get; set; }  // nome do objeto   
    
    // atacar()
    void atacar();
  } // fim da classe


  interface IPosition
  {
    // Controle de posição do objeto
    int X { get; set; }
    int Y { get; set; }
    
    void showpos();     // Mostrar posição do objeto
  
} // fim da interface


Definimos agora uma segunda interface, IPosition, que determina que
seus  objetos  devem  ter propriedades para suportar uma posição 2d
(x,y) e deve mostrar sua posição com o método showpos().
Veja na sequência um exemplo de herança múltipla de interfaces.


  
 class Vampiro:IMonstro, IPosition
 {
 
    // campos particulares do vampiro
    private int m_estrago;   // capacidade de fazer estrago
    private  string m_nome;   // nome do objeto   

    private int xpos;      // posição x do vampiro no mundo
    private int ypos;      // posição y do vampiro no mundo   
    
    // Construtor chama construtor da classe base 
    public Vampiro() 
    {
      m_estrago = 22;
      m_nome = "vampiro";
    }

    // Vamos seguir o contrato da interface IMonstro
    public void atacar()
    {
      string msg = "";
      msg = "\n (Classe Vampiro) {0} te mordeu.";
      Console.Write(msg, m_nome);
      msg = " Vc perdeu {0} pontos de saúde.";
      Console.WriteLine(msg, m_estrago);
    } // atacar().fim 


    // Vamos seguir o contrato da interface IMonstro
    public int Estrago
    {
      set { m_estrago = value; }
      get { return m_estrago; }
    } // fim da propriedade;

    public string Nome
    {
      set { m_nome = value; }
      get { return m_nome; }
    } // fim da propriedade;


    // Vamos seguir o contrato da interface IPosition
    public int X
    {
      set { xpos = value; }
      get { return xpos; }
    } // fim da propriedade;

    public int Y
    {
      set { ypos = value; }
      get { return ypos; }
    } // fim da propriedade;

    public void showpos()
    {
      string spos = "";
      spos = "( " + xpos.ToString() + ", ";
      spos += ypos.ToString() + ")";
      Console.WriteLine(" {0}.posição = {1} \n", m_nome, spos);      
    } // showpos().fim

} // fim da classe


4. Verificando a presença de funcionalidades
É  possível  verificar se  determinado objeto implementa alguma interface
específica. Isso  é  interessante para verificar pela presença de  alguma  
funcionalidade  que  seja  interessante  usar  ou  mesmo  para prevenir o 
uso de métodos inexistentes.



IPosition obj_pos = inimigo[ncx] as IPosition;
if ( obj_pos != null) Console.WriteLine(" IPosition implementada: ");
if (obj_pos != null) obj_pos.showpos();



Neste exemplo, a variável  obj_pos  foi definida como IPosition. Na sequência
jogamos  nela  uma  referência  de  um  objeto existente guardado na variável
inimigo[ncx] como sendo do tipo IPosition. Se o objeto não tiver implementado
a interface IPosition, obj_pos recebe o valor null. Desta  forma  verificamos
se o objeto implementa ou não alguma interface particular.

Segue na sequência o código-fonte do projeto de exemplo deste tópico.

// Projeto prj_interface -  Arquivo: Program.cs
// Ilustra o uso de interfaces
using System;

namespace prj_interface
{
  class Program
  {
    static void Main(string[] args)
    {
      config_janela("projeto prj_interface");

      // Vamos criar uma quadrilha de monstros
      IMonstro godzila = new Ogro();
      Ogro shrek = new Ogro();
      
      Vampiro alucard = new Vampiro();
      alucard.X = 320;
      alucard.Y = 240;

      // Vamos configurar os nomes 
      godzila.Nome = "Godzila";
      shrek.Nome = "Shrek";
      alucard.Nome = "Alucard";

      // Vamos atacar 
      godzila.atacar();
      shrek.atacar();
      alucard.atacar();
      alucard.showpos();
      
      // Vamos criar uma array de monstros!
      IMonstro[] inimigo = new IMonstro[5];
      inimigo[1] = godzila;
      inimigo[2] = shrek;
      inimigo[3] = alucard;

      Console.Write(" ----------------------------------------------------------\n");
      Console.Write(" * Vamos atacar todos de uma vez via inimigo[ncx].atacar()\n");
      Console.WriteLine(" * Também na interface, roda o atacar() da subclasse");
     
      for (int ncx = 1; ncx <= 3; ncx++)
      {
        inimigo[ncx].atacar();

        IPosition obj_pos = inimigo[ncx] as IPosition;
        if ( obj_pos != null) Console.WriteLine(" IPosition implementada: ");
        if (obj_pos != null) obj_pos.showpos();   

      } // fim do for

      Console.ReadKey();
    } // Main().fim

    // Método para configurar a janela
    static void config_janela(String titulo)
    {
      // Vamos configurar a janela
      Console.ForegroundColor = ConsoleColor.Blue;
      Console.BackgroundColor = ConsoleColor.White;
      Console.Title = titulo;
      Console.SetBufferSize(80, 30);
      Console.SetWindowSize(80, 30);
      Console.Clear();
      Console.Write("\n");
    } // config_janela() fim
  } // fim da classe
} // fim do namespace
