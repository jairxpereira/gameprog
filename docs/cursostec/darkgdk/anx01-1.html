Anexo 1.1 Troca remota de arquivos FTP

1 Visão geral

Através das funções de FTP, você pode subir (upload), manipular e descer
(download) arquivos de um servidor remoto utilizando uma conta ftp. Para
testar estes recursos da DarkGdk você precisa ter um ambiente adequado
de servidor ftp.

void dbFTPConnect ( char* szString, char* szString2, char* szString3 );
dbFTPConnect ( sSiteFtp, sUsuario, sSenha);
Esta função faz a conexão com um servidor de arquivos ftp. 
Você precisa informar três strings para fazer a conexão:
o site ftp, usuário e senha.
Utilize a função  dbGetFTPStatus() para verificar se houve 
sucesso ou falha na conexão.

Exemplo de uso: 
dbFTPConnect ("ftp.site.com.br", "usuario", "senha"); 

-------------------------------------------------------------------------------
void dbFTPSetDir( char* dwString );
dbFTPSetDir( sPasta);
Essa função estabelece a pasta ou o diretório corrente de uma 
conexão ftp.

Exemplo:
dbFTPSetDir("//images");

-------------------------------------------------------------------------------
void dbFTPFindFirst ( void );
dbFTPFindFirst();
Essa função aponta para o primeiro arquivo numa pasta ftp.

Exemplo:
dbFTPFindFirst();

-------------------------------------------------------------------------------
int dbGetFTPStatus ( void );
nSts = dbGetFTPStatus();
Essa função retorna o status da conexão ftp.

	0	- indica falha na conexão
	1	- indica sucesso na conexão
	
Exemplo: 
int cnx_sucesso = dbGetFTPStatus();
if (cnx_sucesso == 1) dbText (10,70, "Conectado");
else dbText (10,70, "Falha na conexao");


-------------------------------------------------------------------------------
void dbFTPFindNext ( void );
dbFTPFindNext();
Essa função localiza o próximo arquivo numa pasta ftp.

Exemplo:
dbFTPFindNext();

-------------------------------------------------------------------------------
void dbFTPGetFile ( char* dwString, char* dwString2 );
dbFTPGetFile ( sArquivo, sDestino);
Essa função descarrega de uma só vez o arquivo apontado por sArquivo,
renomeando-o e copiando-o para o nome e caminho dado por sDestino.

Exemplo:
dbFTPGetFile ( dbGetFTPFileName(), dbGetFTPFileName() ); 
* Descarrega o arquivo corrente da pasta ftp para a pasta corrente
do computador local conservando o nome original do arquivo.

Essa função tem uma segunda sintaxe que apresentamos a seguir:
void dbFTPGetFile ( char* dwString, char* dwString2, int iFlag );
dbFTPGetFile ( sArquivo, sDestino, nBlocoTamanho);
Nessa segunda opção o arquivo é baixado em blocos no tamanho dado 
por nBlocoTamanho.  Nesta  sintaxe você precisa utilizar a função
dbFTPProceed() para descarregar o próximo bloco do arquivo e 
acompanhar o progresso com a função dbGetFTPProgress().

Exemplo: 
 if  (!strcmp(tecla,"4")) 
{
	// Baixe o arquivo em pedaços de 1024 bytes
	dbFTPGetFile ( dbGetFTPFileName(), dbGetFTPFileName(),1024  );

	for (; dbGetFTPProgress()!= -1;)
	{        	
	dbText(10,320, dbStr(dbGetFTPProgress())); 
	dbFTPProceed ();
	} //endfor
} // endif 

-------------------------------------------------------------------------------
void dbFTPProceed( void ); 
dbFTPProceed ();
Essa função avança para o próximo bloco de dados do arquivo que
está sendo baixado gradualmente pela função dbFTPGetFile().

Exemplo: 
 if  (!strcmp(tecla,"4")) 
{
	// Baixe o arquivo em pedaços de 1024 bytes
	dbFTPGetFile ( dbGetFTPFileName(), dbGetFTPFileName(),1024  );

	for (; dbGetFTPProgress()!= -1;)
	{        	
	dbText(10,320, dbStr(dbGetFTPProgress())); 
	dbFTPProceed ();
	} //endfor
} // endif 

-------------------------------------------------------------------------------
void dbFTPPutFile ( char* dwString );
dbFTPPutFile ( sArquivo);
Essa função sobe (upload) um arquivo para uma pasta ftp.

Exemplo: 
dbFTPPutFile ("tst.html"); 
 
-------------------------------------------------------------------------------
int dbGetFTPFileType ( void );
nTipo = dbGetFTPFileType(); 
Essa função retorna o tipo de arquivo encontrado numa pasta ftp.

	0	- Indica arquivo comum
	1	- Indica uma pasta ou diretório
	-1	- Indica que a busca de arquivos chegou ao final
	
Exemplo: 
dbFTPFindFirst();
int ntipo = dbGetFTPFileType() 
 
 -------------------------------------------------------------------------------
int dbGetFTPFileSize ( void );
int nTamanho = dbGetFTPFileSize();
Essa função retorna o tamanho do arquivo encontrado. 

Exemplo: 
dbFTPFindFirst();
int ntam = dbGetFTPFileSize() 
 
 -------------------------------------------------------------------------------
char*	dbGetFTPFileName ( void );
sNome = dbGetFTPFileName();
Essa função retorna o nome do arquivo localizado numa pasta ftp.

Exemplo: 
dbFTPFindFirst();
char *snome = dbGetFTPFileName() 
 
 -------------------------------------------------------------------------------
int dbGetFTPProgress ( void );
nPercentual = dbGetFTPProgress();
Essa função retorna o percentual de progresso da operação de download 
gradual de arquivo realizado pela função dbFTPGetFile().
Essa função retorna -1 caso o download tenha sido completado.

Exemplo: 
 if  (!strcmp(tecla,"4")) 
{
	// Baixe o arquivo em pedaços de 1024 bytes
	dbFTPGetFile ( dbGetFTPFileName(), dbGetFTPFileName(),1024  );

	for (; dbGetFTPProgress()!= -1;)
	{        	
	dbText(10,320, dbStr(dbGetFTPProgress())); 
	dbFTPProceed ();
	} //endfor
} // endif 

 -------------------------------------------------------------------------------
char*	dbGetFTPDir	( void ); 
sPasta = dbGetFTPDir(); 
Essa função retorna a pasta selecionada de uma conexão ftp.

Exemplo: 
char *spasta = dbGetFTPDir(); 

Nota: essa função não funcionou em nossos testes.

Para suprimir a ausência dessa funcionalidade você pode utilizar a 
função dbGetFTPFileType() para monitorar o  tipo de  arquivo, sendo 
que o tipo 1 identifica uma pasta. Você deve  monitorar  também  as 
configurações de pasta que se faz com a função dbFTPSetDir().

 
 -------------------------------------------------------------------------------
int dbGetFTPFailure ( void );
nSts = dbGetFTPFailure();
Essa função retorna 1 caso tenha ocorrido algum erro em alguma outra
função ftp. Zero (0) indica ausência de erros.

Exemplo: 
if ( dbGetFTPFailure() == 1)
	dbText (320,240,  dbGetFTPError() ); 	
 
 -------------------------------------------------------------------------------
char*	dbGetFTPError( void );
sInfo = dbGetFTPError();
Essa função retorna uma string trazendo detalhes do erro
ocorrido em uma outra função ftp.

Exemplo: 
if ( dbGetFTPFailure() == 1)
	dbText (320,240,  dbGetFTPError() ); 				
 
 -------------------------------------------------------------------------------
void	dbFTPDisconnect( void );
dbFTPDisconnect();
Essa função encerra a conexão ftp.

Exemplo:
dbGetFTPError();

Veja agora o programa completo:



// tst_ftp.cpp 
// Esse programa ilustra acesso a arquivos remoto via protocolo FTP
#include "DarkGDK.h"
void initsys();
void tst_ftp();
void showlist (char *slista[], int nlista, int xcol, int ylin); 
const int branco = 0xffffff;
const int azul = 0x0000ff;

// the main entry point for the application is this function
void DarkGDK ( void )
{
    // turn on sync rate and set maximum rate to 60 fps
    initsys();
    tst_ftp();
     
    // our main loop
    while ( LoopGDK ( ) )
    {           
        dbSync ( );
    }

    // return back to windows
    return;
} // DarkGdk().end


void initsys() {
dbSyncOff();
dbSetDisplayMode(640,480,32);
dbSetWindowOn();
dbInk (azul, branco);
dbCLS(branco);
dbPositionMouse (1,1);
dbSetWindowTitle("tst_ftp.cpp");
dbSetWindowPosition(320,50);
dbText (10,70,"Aguarde. Conectando ftp...");
dbSync();

} // initsys().end

void tst_ftp() {

bool lUploaded = false;
int cnx_sucesso = 0;

char *slista[] = {"1- ftp find next", "2- download full size", 
"3- ftp find first", "4- download each 1024 bytes", "q- break ",
"6- upload file"};

dbFTPConnect ("ftp.site.com.br", "usuario", "senha");
dbFTPSetDir("//images");
dbFTPFindFirst();

cnx_sucesso = dbGetFTPStatus();
dbSetTextOpaque();
dbText (10,70, dbSpaces(40));
if (cnx_sucesso == 1) dbText (10,70, "Conectado");
else dbText (10,70, "Falha na conexao");


while (dbMouseClick() != 2) {

showlist (&slista[0], 6, 200, 10);

char *tecla = dbInKey();

if (!strcmp(tecla,"1")) {
dbFTPFindNext();
dbSleep (70);
}//endif

if (!strcmp(tecla,"2")) {
dbFTPGetFile ( dbGetFTPFileName(), dbGetFTPFileName() );
} //endif


if (!strcmp(tecla,"3"))dbFTPFindFirst();

    if  (!strcmp(tecla,"4")) 
    {
    // download every 1024 bytes
        dbFTPGetFile ( dbGetFTPFileName(), dbGetFTPFileName(),1024  );

        for (; dbGetFTPProgress()!= -1;)
        {        
        
        dbText(10,280, "dbGetFTPProgress()");
        dbText(10,300, "-------------------");
        // Limpa os dados anteriores
        dbText(10,320, dbSpaces(40));
        dbText(10,320, dbStr(dbGetFTPProgress())); 
        dbFTPProceed ();
        if (dbUpKey()==1) break;
        dbSleep (100);
        } //endfor
    } // endif

if  (!strcmp(tecla,"q"))break;

    if  (!strcmp(tecla,"6")) {
        //upload file
        if (!lUploaded) dbFTPPutFile ("tst.html");
        if (!lUploaded) dbPrint ("ok"); 
        lUploaded = true;   
    } // endif
  

// Pega tipo de arquivo: 0:arquivo 1:Dir
dbText(10,200, "dbGetFTPFileType()");
dbText(10,220, "-------------------");
// Limpa os dados anteriores
dbText(10,240, dbSpaces(40));
dbText(10,240, dbStr(dbGetFTPFileType())); 

// Pega tamanho do arquivo
dbText (200,200, "dbGetFTPFileSize()" );
dbText (200,220, "------------------" );
// Limpa os dados anteriores
dbText (200,240, dbSpaces(40));
dbText (200,240,dbStr (dbGetFTPFileSize())); 

// Pega nome do arquivo
dbText (390,200, "dbGetFTPFileName()" );
dbText (390,220, "------------------" );
// Limpa os dados anteriores
dbText (390,240, dbSpaces(40));
dbText (390,240, dbGetFTPFileName()); 

// Pega progresso do download: -1 completo
dbText(10,280, "dbGetFTPProgress()");
dbText(10,300, "-------------------");
// Limpa os dados anteriores
dbText(10,320, dbSpaces(40));
dbText(10,320, dbStr(dbGetFTPProgress())); 

// Pega status da conexão 1:conectado 0:desconectado
dbText(200,280, "dbGetFTPStatus()");
dbText(200,300, "-------------------");
// Limpa os dados anteriores
dbText(200,320, dbSpaces(40));
dbText(200,320, dbStr(dbGetFTPStatus())); 

// Obtém nome da pasta
dbText(390,280, "dbGetFTPDir()");
dbText(390,300, "-------------------");
// Limpa os dados anteriores
dbText(390,320, dbSpaces(40));
dbText(390,320, dbGetFTPDir()); 

// Verifica se houve erro: 0:Ok 1:erro
dbText(10,360, "dbGetFTPFailure()");
dbText(10,380, "-------------------");
// Limpa os dados anteriores
dbText(10,400, dbSpaces(40));
dbText(10,400, dbStr(dbGetFTPFailure())); 

// Pega nome do erro
dbText(200,360, "dbGetFTPError()");
dbText(200,380, "-------------------");
// Limpa os dados anteriores
dbText(200,400, dbSpaces(40));
dbText(200,400, dbGetFTPError()); 

} // endwhile

// Desconecta
dbFTPDisconnect();
dbCLS(branco); dbCenterText (320,240, "Pressione ESC para sair.");
dbWaitKey();

dbSync();

} // endfunc: tst_ftp()


void showlist (char *slista[], int nlista, int xcol, int ylin) {
//*** showlist - Mostras lista de opções

int ncx = 0;
char *stemp;

int leapline = 20;

dbSetTextOpaque();

for (ncx = 0; ncx < nlista; ncx++)
{
 stemp = slista[ncx];
 dbInk (azul, branco);
 dbText (xcol, ylin, stemp);
 ylin = ylin + leapline;
 dbSync();
 } // endfor
} // endfunc: showlist
