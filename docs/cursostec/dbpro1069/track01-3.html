01.3 Definindo arrays

1. Arrays 
Uma array pode ser visualizada como uma lista de dados ou uma tabela
de dados. Para criar uma array de um certo tipo de dados é utilizado
a instrução dim da seguinte forma:

	dim lista(100) as integer
	
Na linha acima nós criamos uma lista com 100 inteiros que vão ser 
acessados conforme o seguinte exemplo: 

	lista(1) = 320
	x = lista(88) 
	
Veja outro exemplo de uma lista de strings:

	dim fase_nome(10) as string
	fase_nome(1) = "Vale do Godzila"
	fase_nome(2) = "Encontro com Dracula"
	fase_nome(3) = "Combate final" 
	
	
2. Array bidimensional 
Para criar uma array representando uma tabela de dados com linhas e
colunas proceda da seguinte forma:

	dim pos(5,2) as integer 
	
A linha acima cria uma tabela com 5 linhas e 2 colunas que vão ser
manipuladas desse jeito:

	pos(1,1) = 320
	pos(1,2) = 240
	xpos = tabela(1,1)
	ypos = tabela(1,2) 
	
	
3. Array de estruturas
Naturalmente é possível criar arrays de estruturas. Veja o seguinte
exemplo: 

' Estrutura Monstro
type Monstro
   nome    as string
   tipo    as byte
energia    as integer
estrago    as byte
endtype

' Declaração a array de uma estrutura
dim inimigos(10)       as Monstro

' Configurando
' ----------------------------------------
inimigos(1).nome = "Godzila"
inimigos(1).tipo = 1
inimigos(1).energia = 100
inimigos(1).estrago = 6 

Veja agora o programa exemplo:



 





' Programa arrays.dba
' Ilustra o uso de array

' Configurando a janela
branco as integer = 0xFFFFFF
azul   as integer = 0x0000FF

' Estrutura Monstro
type Monstro
   nome    as string
   tipo    as byte
energia    as integer
estrago    as byte
endtype

' Estrutura Ponto3D
type Ponto3D
x as float
y as float
z as float
endtype

' Variáveis de trabalho
monstro_qtd   as integer = 3
ncx           as integer = 0
linha         as integer = 1


' Declaração de arrays
dim inimigos(10)       as Monstro
dim fase_nome(10)      as string
dim monstro_pos(100)   as Ponto3D
dim tabela_pos(10,3)   as integer


' Configurando a janela
set window on
set window title "arrays.dba"
ink azul, branco
cls branco


' Configurando nome das fases
' ----------------------------------------
fase_nome(1) = "Vale do Godzila"
fase_nome(2) = "Encontro com Dracula"
fase_nome(3) = "Combate final"


' Configurando 3 monstros
' ----------------------------------------
inimigos(1).nome = "Godzila"
inimigos(1).tipo = 1
inimigos(1).energia = 100
inimigos(1).estrago = 6

inimigos(2).nome = "Dracula"
inimigos(2).tipo = 3
inimigos(2).energia = 8
inimigos(2).estrago = 10

inimigos(3).nome = "Boss final"
inimigos(3).tipo = 10
inimigos(3).energia = 1000
inimigos(3).estrago = 15


' Configurando posição dos monstros
' ----------------------------------------
monstro_pos(1).x = 320
monstro_pos(1).y = 240
monstro_pos(1).z = 100

monstro_pos(2).x = 400
monstro_pos(2).y = 249
monstro_pos(2).z = 100

monstro_pos(3).x = 400
monstro_pos(3).y = 240
monstro_pos(3).z = 100

' Guardando posição dos monstros na tabela
' ----------------------------------------
tabela_pos(1,1) = monstro_pos(1).x
tabela_pos(1,2) = monstro_pos(1).y
tabela_pos(1,3) = monstro_pos(1).z

tabela_pos(2,1) = monstro_pos(2).x
tabela_pos(2,2) = monstro_pos(2).y
tabela_pos(2,3) = monstro_pos(2).z

tabela_pos(3,1) = monstro_pos(3).x
tabela_pos(3,2) = monstro_pos(3).y
tabela_pos(3,3) = monstro_pos(3).z


' Monstrando status dos monstros
for ncx = 1 to monstro_qtd
print
print " Nome da fase: ", fase_nome(ncx)
print " Nome:    ", inimigos(ncx).nome
print " Energia: ", inimigos(ncx).energia
print " Tipo :   ", inimigos(ncx).tipo
print " Estrago: ", inimigos(ncx).estrago
print " Posicao do montro:";
print "(",monstro_pos(ncx).x," ", monstro_pos(ncx).y, " ", monstro_pos(ncx).z, ")"
next ncx

' Mostrando a tabela de posições dos montros
print
print " Tabela de posicoes dos monstros:"
for linha = 1 to monstro_qtd
print "(",tabela_pos(linha,1)," ",tabela_pos(linha,2), " ", tabela_pos(linha, 3), ")"
next linha

suspend for key
