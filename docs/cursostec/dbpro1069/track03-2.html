03-2. Execução condicional 2/2

select case

 A construção select case permite a execução condicional de um bloco de código
 conforme a verificação de uma condição.

 Sintaxe de uso: 

 select var
        case valor
        ... código ...
        endcase

        case default
        ... código ...
        endcase
 endselect 

 A variável pode ser do tipo real, inteiro ou string.

 Veja o exemplo de uso:




cls

print

input "Digite sua fruta preferida: ", fruta$

select  fruta$

        case "banana"
        print "A banana custa r$ 100"
        endcase

        case "goiaba"
        print "A goiaba custa r$ 50"
        endcase

        case default
        print "Não temos essa fruta no momento!"
        endcase

endselect

suspend for key



 Aqui nesse nosso exemplo o código que será executado depende da informação
 entrada.

 A instrução case default é opcional e o seu bloco de código somente é
 executado quando nenhuma das condições é verificada como verdadeira.
 
 Veja agora o programa exemplo:


 





Rem Project: switch
Rem Created: 26/07/2013 21:20:21
Rem ***** Main Source File *****

' Configurando a janela
branco as integer = 0xFFFFFF
azul   as integer = 0x0000FF
set window on
set window title "switch.dba"
ink azul, branco
cls branco

' Itens do inventário do jogador
espada_qtd     as integer = 0
escudo_qtd     as integer = 0
flecha_qtd     as integer = 0

' Preço dos itens na lojinha do game
espada_custo   as integer = 120
escudo_custo   as integer = 60
flecha_custo   as integer = 8

' Conta (R$) do jogador e monitoramento da escolha
conta_jogador  as integer = 300
escolha        as integer = 0

do
cls branco
print

set cursor 20,10
print " Dinheiro: ", conta_jogador, " Espadas: ", espada_qtd;
print " Escudos: ", escudo_qtd, " Flechas: ", flecha_qtd

set cursor 20,40: print "[1] - Espada - R$ 120,00 "
set cursor 20,60: print "[2] - Escudo - R$ 60,00  "
set cursor 20,80: print "[3] - Flecha - R$ 8,00   "
set cursor 20,100: print "[4] - Sair"

set cursor 20,140
input "Digite sua escolha: ", escolha


select escolha
case 1
conta_jogador = conta_jogador - espada_custo
espada_qtd = espada_qtd + 1
endcase

case 2
conta_jogador = conta_jogador - escudo_custo
escudo_qtd = escudo_qtd + 1
endcase

case 3
conta_jogador = conta_jogador - flecha_custo
flecha_qtd = flecha_qtd + 1
endcase

case 4
end
endcase

case default
print
print " Este item nao existe!"
endcase

endselect

print
print " Dinheiro: ", conta_jogador

print " Pressione qualquer tecla para continuar "
suspend for key
loop

end
