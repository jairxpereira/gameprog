06-4 Classe abstrata
1. Visão geral
A classe abstrata é desenhada especificamente com a intenção de ser uma
classe base para herança;  ela apenas define os membros que deverão ser
implementados  pelas  subclasses.  Ela  por  si  mesma  não  implementa 
funcionalidades  e  por isso  não pode produzir   objetos   diretamente. 
A classe abstrata apenas estabelece o layout que deve ser seguido pelas
classes que a herdam.

O melhor exemplo de aplicação da classe abstrata vem do sistema Windows
que define uma classe abstrata Controle.  A  classe  estabelece  apenas
a idéia geral do que  é  um  controle:  que  vai  ter  largura,  altura,
que  vai  receber  um  click  e  tratá-lo  e  etc.  A concretização das 
funcionalidades  é  então  somente  feita por botões, janelas, barras 
de rolagem e demais controles do sistema Windows. 

Examinando a concepção da classe abstrata, pode-se resumir que ela 
apenas  estabelece  o  projeto de uma interface comum que todas as
subclasses que a herdam devem possuir. A Microsoft aproveitou essa
pegada  da  classe  abstrata  e embutiu toda essa concepção em uma 
nova  entidade  de  programação criada chamada interface. A classe
abstrata vai  ser explicada aqui por questões de 'completitude' do
curso  mas  vamos  ver  adiante  que  é  mais  vantajoso  produzir  
interfaces do que utilizar classes abastratas.

2. Definindo uma classe abstrata
No processo de construção de uma classe abstrata vai ocorrer o
seguinte:

* Coloque a instrução abstract antes do nome da classe

* Coloque a instrução abstract na assinatura do método

* Na subclasse, coloque a instrução override na assinatura
do método. Na classe base, o método definido como abstract
é automaticamente um método virtual.



abstract class Monstro
{
// atacar()
abstract public void atacar();
} // fim da classe





class Vampiro : Monstro
{


// O Vampiro morde no ataque!
public override void atacar()
{

( código...)

} // atacar().fim 
} // fim da classe



3. Observações interessantes

*  Não é possível instanciar objetos do tipo da classe abstrata. 
Porém, ainda é permitido assinalar a variável como sendo do tipo
da classe abstrata:
	
   Monstro godzila = new Ogro(111);
   Monstro[] inimigo = new Monstro[5]; 

Essa característica garante a continuidade do uso do  polimorfismo
na aplicação. Criamos no projeto exemplo  uma  array  de  Monstros
que receberam em cada elemento um objeto diferente das subsclasses.

* A classe abstrata define os atributos comuns e o construtor dela
ainda é totalmente funcional!:


abstract class Monstro
{

// atributos
public int m_estrago;   // capacidade de fazer estrago
public string m_nome;   // nome do objeto

// Construtor 01 Mostro()
public Monstro()
{

  m_estrago = 10;
  m_nome = "monstro";
} // fim do Construtor 01

// Construtor 02 Monstro (int nval)
public Monstro(int estrago)
{
  m_estrago = estrago;
  m_nome = "monstro";
} // fim do construtor 02


// atacar()
abstract public void atacar();

} // fim da classe

// Projeto prj_clsAbstrata -  Arquivo:Monstro.cs
// Ilustra o uso de classe abstrata
using System;

namespace prj_clsAbstrata
{
  abstract class Monstro
  {

    // atributos
    public int m_estrago;   // capacidade de fazer estrago
    public string m_nome;   // nome do objeto

    // Construtor 01 Mostro()
    public Monstro()
    {

      m_estrago = 10;
      m_nome = "monstro";
    } // fim do Construtor 01

    // Construtor 02 Monstro (int nval)
    public Monstro(int estrago)
    {
      m_estrago = estrago;
      m_nome = "monstro";
    } // fim do construtor 02


    // atacar()
    abstract public void atacar();

  } // fim da classe
} // fim do namespace







// Projeto prj_clsAbstrata -  Arquivo: Ogro.cs
// Ilustra o uso de classe abstrata
using System;

namespace prj_clsAbstrata
{
  class Ogro : Monstro
  {
    // Construtor chama construtor da classe base 
    public Ogro() : base(120) { }
    public Ogro(int nval) : base(nval) { }

    public override void atacar()
    {
      string msg = "";
      Console.Write("\n");

      // Vamos aproveitar o trabalho anterior!
     
      msg = " (Classe Ogro) {0} ataca de novo com pedras e...";
      Console.WriteLine(msg, m_nome);
      msg = " ...vc perde mais {0} pontos de energia.\n";
      Console.WriteLine(msg, m_estrago);
    }// atacar().fim
  } // fim da classe
} // fim do namespace





// Projeto prj_clsAbstrata -  Arquivo: Vampiro.cs 
// Ilustra o uso de classe abstrata
using System;

namespace prj_clsAbstrata
{
  class Vampiro : Monstro
  {
    // Construtor chama construtor da classe base 
    public Vampiro() : base(22) { }

    // O Vampiro morde no ataque!
    public override void atacar()
    {
      string msg = "";
      msg = " (Classe Vampiro) {0} te mordeu.";
      Console.Write(msg, m_nome);
      msg = " Vc perdeu {0} pontos de saúde.\n";
      Console.WriteLine(msg, m_estrago);
    } // atacar().fim 
  } // fim da classe
} // fim do namespace





// Projeto prj_clsAbstrata -  Arquivo: Program.cs
// Ilustra o uso de classe abstrata
using System;

namespace prj_clsAbstrata
{
  class Program
  {
    static void Main(string[] args)
    {
      config_janela("projeto prj_clsAbstrata");

      // Vamos criar uma quadrilha de monstros
      Monstro godzila = new Ogro(111);
      Ogro shrek = new Ogro();
      Vampiro alucard = new Vampiro();

      // Vamos configurar os nomes 
      godzila.m_nome = "Godzila";
      shrek.m_nome = "Shrek";
      alucard.m_nome = "Alucard";

      // Vamos atacar 
      godzila.atacar();
      shrek.atacar();
      alucard.atacar();

      // Vamos criar uma array de monstros!
      Monstro[] inimigo = new Monstro[5];
      inimigo[1] = godzila;
      inimigo[2] = shrek;
      inimigo[3] = alucard;

      Console.Write("----------------------------------------------------------\n");
      Console.Write(" *Vamos atacar todos de uma vez via inimigo[ncx].atacar()\n");
      Console.WriteLine("*Com virtual\\override agora roda o atacar() da subclasse");
      for (int ncx = 1; ncx <= 3; ncx++) inimigo[ncx].atacar();

      Console.ReadKey();
    } // Main().fim

    // Método para configurar a janela
    static void config_janela(String titulo)
    {
      // Vamos configurar a janela
      Console.ForegroundColor = ConsoleColor.Blue;
      Console.BackgroundColor = ConsoleColor.White;
      Console.Title = titulo;
      Console.SetBufferSize(80, 30);
      Console.SetWindowSize(80, 30);
      Console.Clear();
      Console.Write("\n");
    } // config_janela() fim
  } // fim da classe
} // fim do namespace
