11-2 Reflexão
1. Visão geral

A imagem acima diz muito sobre o conceito de reflexão. Temos acima os
vários membros e métodos da classe System.Console. Neste tópico vamos
aprender como vasculhar as informações da vasta biblioteca .net.

O núcleo básico do framework .net está contido no arquivo mscorlib.dll.
Este arquivo e outros em condições semelhantes é chamado de assembly e
pode ser resumido como o recipiente primordial aonde está  contido  as
dezenas de tipos que a linguagem  C#  pode  produzir  como  namespaces,
classes, propriedades, eventos e etc.

Os exemplos desse tópico fazem uso do namespace System.Reflection.

2.1 Vasculhando uma assembly



// Carrega a assembly
Assembly lib = Assembly.Load("mscorlib");

// Obtém os diversos tipos dentro da assembly (classes, interfaces, etc)
Type[] entidades = lib.GetTypes();

// Verifica a quantidade de tipos dentro da assembly
int ntam = entidades.Length;

// Exibe os diversos tipos existentes
for (int ncx = 0; ncx < ntam; ncx++)
{
Type entidade = entidades[ncx];
Console.WriteLine("\t {0}", entidade.ToString());
}


Estas poucas linhas mostram mais de 2300 tipos presentes em  mscorlib:
classes, enumerações, interfaces e outros tipos. Mas isso é só a ponta
do  iceberg  porque  dentro das classes tem os métodos, os campos, os 
eventos e demais membros.

Assembly lib = Assembly.Load("mscorlib");
Esta linha carrega a assembly especificada produzindo o objeto lib
desse tipo.

Type[] entidades = lib.GetTypes();
O próximo passo é conseguir uma array de objetos tipo Type retornado
por lib.GetTypes() do objeto assembly instanciado.

Na sequência utilize for ou foreach para manipular cada objeto Type
dentro da coleção retornada ( entidades ).

2.2.1 Pegando um tipo específico



// Mostra membros presentes em System.Console
Type console = lib.GetType("System.Console");
MemberInfo[] membros = console.GetMembers();
foreach (MemberInfo mi in membros)
{
// Exibe membro e tipo do membro
Console.Write(" {0}", mi.MemberType);
Console.Write(" \t{0}\n", mi.ToString());
} // endforeach



Type console = lib.GetType("System.Console");
Utilize GetType() do objeto assembly (lib) para pegar um tipo específico. 

MemberInfo[] membros = console.GetMembers();
Na sequência utilize um  objeto  MemberInfo[]  para  receber  os  membros 
passados por GetMembers() do tipo específico que foi obtido (console).

Utilizamos aqui GetMembers() do objeto console para pegar todos os seus
membros.  Neste ponto com pouca variação no código, utilizando qualquer 
outra função que começa com Get* desse objeto pode-se vasculhar métodos,
eventos, propriedades e outros itens: GetMethods(), GetEvents(), etc.

2.2.2 Pegando membros com critérios específicos


// Mostra métodos dentro de certos critérios de System.Console
membros = console.FindMembers(MemberTypes.Method, 
BindingFlags.Public | BindingFlags.Static,
Type.FilterNameIgnoreCase, "get*");

foreach (MemberInfo mi in membros)
{
 // Exibe membro e tipo do membro
 Console.Write(" {0}", mi.MemberType);
 Console.Write(" \t{0}\n", mi.ToString());
} // endforeach


Com o método FindMembers() do objeto console é possível selecionar membros
estabelecendo critérios específicos. Por exemplo, podemos selecionar todos
os membros ou métodos que comecem com a palavra Get  ou  get.  Observe  no
exemplo acima como o  contexto foi configurado para obter todos os métodos
que  comecem  com  a  palavra get*  ignorando o case (caixa alta\baixa) do
nome do método.

3. Código fonte do projeto de exemplo


// Projeto prj_Reflection01 - Arquivo: Program.cs
// Mostra como usar System.Reflection
using System;
using System.Reflection;

namespace prj_Reflection01
{
  class Program
  {

    public static void Main()
    {
      config_janela("prj_Reflection01");

      // /campos para contagem dos itens
      int qtd_classes = 0;
      int qtd_interfaces = 0;
      int qtd_enums = 0;

      int linha = 0;

      // Carrega a assembly
      Assembly lib = Assembly.Load("mscorlib");

      // Obtém os diversos tipos dentro da assembly (classes, interfaces, etc)
      Type[] entidades = lib.GetTypes();

      // Verifica a quantidade de tipos dentro da assembly
      int ntam = entidades.Length;

      // Exibe os diversos tipos existentes
      Console.WriteLine(" Tipos presentes em mscorlib");
      Console.WriteLine(" --------------------------------------------------");
      int nlimite = 25;
      for (int ncx = 0; ncx < nlimite; ncx++)
      {
        Type entidade = entidades[ncx];
        Console.WriteLine("\t {0}", entidade.ToString());
        // Contagem de classes, interfaces e enumerações
        if (entidade.IsInterface) qtd_interfaces++;
        if (entidade.IsClass) qtd_classes++;
        if (entidade.IsEnum) qtd_enums++;
      } //endfor each

      // Exibe algumas estatísticas
      Console.WriteLine("\n\n Contagem: \n Classes      #:{0}", qtd_classes);
      Console.WriteLine(" Interfaces           #:{0}", qtd_interfaces);
      Console.WriteLine(" Enumerações          #:{0}", qtd_enums);
      Console.WriteLine(" Total de entidades   #:{0}\n\n", ntam);

      pause();
      // Mostra membros presentes em System.Console
      Type console = lib.GetType("System.Console");
      MemberInfo[] membros = console.GetMembers();
      Console.WriteLine("\n Membros presentes em System.Console");
      Console.WriteLine(" --------------------------------------------------");
      foreach (MemberInfo mi in membros)
      {

        // Exibe membro e tipo do membro
        Console.Write(" {0}", mi.MemberType);
        Console.Write(" \t{0}\n", mi.ToString());

        // Pula uma linha pra facilitar a leitura
        linha++;
        if (linha % 5 == 0) Console.Write("\n");
        if (linha % 20 == 0) pause();

      } // endforeach

      pause();
      // Mostra métodos presentes em System.Console
      membros = console.GetMethods();
      Console.WriteLine("\n Métodos presentes em System.Console");
      Console.WriteLine(" --------------------------------------------------");
      foreach (MemberInfo mi in membros)
      {

        // Exibe membro e tipo do membro
        Console.Write(" {0}", mi.MemberType);
        Console.Write(" \t{0}\n", mi.ToString());

        // Pula uma linha pra facilitar a leitura
        linha++;
        if (linha % 5 == 0) Console.Write("\n");
        if (linha % 20 == 0) pause();

      } // endforeach


      pause();
      // Mostra métodos dentro de certos critérios de System.Console
      membros = console.FindMembers(MemberTypes.Method, 
        BindingFlags.Public | BindingFlags.Static,
        Type.FilterNameIgnoreCase, "get*");

      Console.WriteLine("\n Métodos que começam com get...");
      Console.WriteLine(" --------------------------------------------------");
      foreach (MemberInfo mi in membros)
      {
        // Exibe membro e tipo do membro
        Console.Write(" {0}", mi.MemberType);
        Console.Write(" \t{0}\n", mi.ToString());
        // Pula uma linha pra facilitar a leitura
        linha++;
        if (linha % 5 == 0) Console.Write("\n");
        if (linha % 20 == 0) pause();
      } // endforeach

      pause(); pause();

      
    } // Main().fim

    
    // Pausa para a leitura
    private static void pause()
    {
      Console.Read();
    } // pause().fim


    // Método para configurar a janela
    private static void config_janela(string titulo)
    {
      // Vamos configurar a janela
      Console.ForegroundColor = ConsoleColor.Blue;
      Console.BackgroundColor = ConsoleColor.White;
      Console.Title = titulo;
      Console.SetBufferSize(120, 2345);
      Console.SetWindowSize(120, 40);
      Console.Clear();
      Console.Write("\n");
    } // config_janela() fim   
  } // fim da classe Program
} // fim do namespace
