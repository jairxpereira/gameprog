28.01 ftime
        Esta função obtém o tempo corrente, preenchendo uma struct timeb.
        
        void ftime (struct time *buf);
        ftime (&xLocal);

        xLocal.....É uma instância da struct timeb

        Elementos da struct timeb:
        ------------------------------------------------------
        long xLocal.time ........Tempo em segundo decorrido desde 01/01/1970 0:00hs
        short xLocal.millitm ..........Milessegundos
        short xLocal.timezone .........Diferença de fuso horário entre GMT e o local em minutos
        short xLocal.dstflag ..........Horário de verão

        Ex.:    struct timeb xLocal;
                char *sTempo;
                ftime(&xLocal);
                sTempo = ctime(&xLocal.time);
                printf ("sTempo: %s\n\n",sTempo);
                /* Resultou em sTempo: Sun Feb 06 09:50:08 2005 */

        Veja o programa exemplo: 
        #include "stdio.h"
        #include "conio.h"
        #include "sys\timeb.h"

        int main(void) {
        struct timeb xLocal;
        char *sTempo;

        ftime(&xLocal);
        sTempo = ctime(&xLocal.time);

        clrscr();
        printf ("sTempo: %s\n\n",sTempo);
        printf ("xLocal.time: %li\n",xLocal.time);
        printf ("xLocal.millitm: %d\n",xLocal.millitm);
        printf ("xLocal.timezone: %d\n",xLocal.timezone);
        printf ("xLocal.dstflag: %d\n",xLocal.dstflag);

        puts("End of Program.\n");
        getch();
        return 0; }

        Saída:
        sTempo: Sun Feb 06 09:50:08 2005

        xLocal.time: 1107701408
        xLocal.millitm: 920
        xLocal.timezone: 300
        xLocal.dstflag: 0
        End of Program.
